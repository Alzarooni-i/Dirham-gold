function t(key){ return text[lang][key]; }


function toggleLanguage() {

  lang = lang === "ar" ? "en" : "ar";

  document.documentElement.lang = lang;

  document.documentElement.dir = t("dir");


  document.getElementById("langBtn").innerText = t("btn");

  document.getElementById("subtitle").innerText = t("subtitle");

  document.getElementById("globalTitle").innerText = t("globalTitle");

  document.getElementById("ounce").placeholder = t("ouncePH");

  document.getElementById("usd").placeholder = t("usdPH");

  document.getElementById("readyText").innerText = t("ready");

  document.getElementById("gramTitle").innerText = t("gramTitle");

  document.getElementById("normalTitle").innerText = t("normalTitle");

  document.getElementById("weight").placeholder = t("weightPH");

  document.getElementById("making").placeholder = t("makingPH");

  document.getElementById("reverseTitle").innerText = t("reverseTitle");

  document.getElementById("rWeight").placeholder = t("weightPH");

  document.getElementById("total").placeholder = t("totalPH");

  document.getElementById("reverseBtn").innerText = t("reverseBtn");


  updatePrices();

}


function getGramPrice(k) {

  const ounce = parseFloat(document.getElementById("ounce").value) || 0;

  const usd = parseFloat(document.getElementById("usd").value) || 3.678;

  const gram24 = (ounce / 31.1035) * usd;

  return gram24 * (k / 24);

}


function updatePrices() {

  const ounce = document.getElementById("ounce").value;

  const prices = document.getElementById("prices");


  if (!ounce) {

    prices.innerText = t("waitingPrices");

    return;

  }


  prices.innerHTML = "";

  [24,22,21,18].forEach(k => {

    prices.innerHTML += `

      <div class="row">

        <span>${k}K</span>

        <span>${getGramPrice(k).toFixed(2)} ${t("aed")}</span>

      </div>

    `;

  });

}


function normalCalc() {

  const weight = parseFloat(document.getElementById("weight").value) || 0;

  const karat = parseFloat(document.getElementById("karat").value);

  const making = parseFloat(document.getElementById("making").value) || 0;


  const total = ((getGramPrice(karat) + making) * weight) * 1.05;


  document.getElementById("normalResult").innerText =

    `${t("final")}: ${total.toFixed(2)} ${t("aed")}`;

}


function reverseCalc() {

  const weight = parseFloat(document.getElementById("rWeight").value) || 0;

  const karat = parseFloat(document.getElementById("rKarat").value);

  const total = parseFloat(document.getElementById("total").value) || 0;


  if (weight <= 0) {

    document.getElementById("reverseResult").innerText = t("waitingCalc");

    return;

  }


  const beforeVAT = total / 1.05;

  const goldOnly = getGramPrice(karat) * weight;

  const makingPerGram = (beforeVAT - goldOnly) / weight;


  document.getElementById("reverseResult").innerText =

    `${t("making")}: ${makingPerGram.toFixed(2)} ${t("aed")}`;

}
