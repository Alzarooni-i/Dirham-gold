<script>


let currentLang = "en";


function gramPrice(karat){


let ounce =

parseFloat(document.getElementById("ounce").value) || 0;


let usd =

parseFloat(document.getElementById("usd").value) || 3.6725;


let gram24 = (ounce / 31.1035) * usd;


return gram24 * (karat / 24);

}


function updatePrices(){


let html = "";


[24,22,21,18].forEach(k=>{


html += `

<div>

${k}K : ${gramPrice(k).toFixed(2)} AED

</div>

`;


});


document.getElementById("prices").innerHTML = html;

}


function calculatePrice(){


let weight =

parseFloat(document.getElementById("weight").value) || 0;


let karat =

parseFloat(document.getElementById("karat").value);


let making =

parseFloat(document.getElementById("making").value) || 0;


let gold = gramPrice(karat);


let subtotal = (gold + making) * weight;


let final = subtotal * 1.05;


document.getElementById("finalResult").innerHTML =

`Final Price: ${final.toFixed(2)} AED`;

}


function calculateReverse(){


let weight =

parseFloat(document.getElementById("reverseWeight").value) || 0;


let karat =

parseFloat(document.getElementById("reverseKarat").value);


let total =

parseFloat(document.getElementById("totalPrice").value) || 0;


let beforeVat = total / 1.05;


let goldOnly = gramPrice(karat) * weight;


let makingTotal = beforeVat - goldOnly;


let makingGram = makingTotal / weight;


document.getElementById("reverseResult").innerHTML =

`Making Charge: ${makingGram.toFixed(2)} AED per gram`;

}


document.getElementById("ounce").addEventListener("input",updatePrices);


function toggleLanguage(){


if(currentLang==="en"){


currentLang="ar";


document.documentElement.lang="ar";

document.body.style.direction="rtl";


document.getElementById("subtitle").innerText =

"حاسبة الذهب الإماراتية الكاملة";


document.getElementById("ounceTitle").innerText =

"سعر الأونصة العالمي";


document.getElementById("normalTitle").innerText =

"الحساب العادي";


document.getElementById("reverseTitle").innerText =

"الحساب العكسي للمصنعية";


document.getElementById("calcBtn").innerText =

"احسب السعر النهائي";


document.getElementById("reverseBtn").innerText =

"احسب المصنعية";


document.getElementById("langBtn").innerText =

"English";


}else{


currentLang="en";


location.reload();


}


}
